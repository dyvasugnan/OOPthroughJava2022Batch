import java.util.Scanner;
public class Percentage{
    public static void main(String args[]){
        int maths;
        int physics;
        int chemistry;
        float percentage;
        Scanner sc = new Scanner(System.in);
        System.out.print("maths marks are:");
        maths = sc.nextInt();
        System.out.print("physics marks are:");
        physics = sc.nextInt();
        System.out.print("chemistry marks are:");
        chemistry = sc.nextInt();
        percentage = ((maths+physics+chemistry)*100/300);
        System.out.print("percentage of the student is"+percentage);
        
        
        
        
        
    }
}
